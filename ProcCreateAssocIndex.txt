Alter Procedure ProcCreateAssocIndex

As

truncate table tblAssocIndex

/* Fill tblAssocIndex with the fields from 2 qry's used to 
get the association names based on the data exported into tblAssocPages from LinkUP. 
This new table will be used to generate the Assoc Index*/

Insert into tblAssocIndex ([Name], page_no)
SELECT assoc_name,assoc_page_no
FROM   qryAssocMainIndex

Insert into tblAssocIndex ([Name], page_no)
SELECT sub_name, sub_page_no
FROM  qryAssocSubIndex


Return
